SVG5 制作过程详解

探索三原色混合交互的完整制作流程

1

创建画布和四个圆形

首先创建一个600x400像素的SVG画布，并绘制四个不同位置和半径的圆形。三个圆形代表三原色（红、绿、蓝），一个圆形在中心位置用于显示颜色混合结果。

<!-- 创建SVG画布和四个圆形 -->
<svg id="svg5" xmlns="http://www.w3.org/2000/svg" width="100%" height="100%" viewBox="0 0 600 400">
  <!-- 背景 -->
  <rect x="0" y="0" width="600" height="400" fill="#f8f9fa"/>

  <!-- 混合结果圆 -->
  <circle id="mixCircle" cx="300" cy="200" r="80" fill="rgb(128,128,128)" stroke="#333" stroke-width="2"/>

  <!-- 三原色圆 -->
  <circle id="redCircle" cx="150" cy="120" r="40" fill="url(#redGradient)" stroke="#cc0000" stroke-width="3"/>
  <circle id="greenCircle" cx="150" cy="280" r="40" fill="url(#greenGradient)" stroke="#00cc00" stroke-width="3"/>
  <circle id="blueCircle" cx="450" cy="200" r="40" fill="url(#blueGradient)" stroke="#0000cc" stroke-width="3"/>
</svg>
2

安排三原色圆的位置和颜色

将三个原色圆围绕在中心混合圆周围，分别位于左上（红色）、左下（绿色）和右侧（蓝色）。为每个原色圆添加渐变效果，使其看起来更加立体和生动。

红色 (R)
绿色 (G)
蓝色 (B)
<!-- 定义渐变 -->
<defs>
  <linearGradient id="redGradient" x1="0%" y1="0%" x2="100%" y2="100%">
    <stop offset="0%" stop-color="#ff0000" stop-opacity="0.8"/>
    <stop offset="100%" stop-color="#ff6666" stop-opacity="0.8"/>
  </linearGradient>

  <!-- 类似定义绿色和蓝色渐变 -->
  <linearGradient id="greenGradient" x1="0%" y1="0%" x2="100%" y2="100%">
    <stop offset="0%" stop-color="#00ff00" stop-opacity="0.8"/>
    <stop offset="100%" stop-color="#66ff66" stop-opacity="0.8"/>
  </linearGradient>

  <linearGradient id="blueGradient" x1="0%" y1="0%" x2="100%" y2="100%">
    <stop offset="0%" stop-color="#0000ff" stop-opacity="0.8"/>
    <stop offset="100%" stop-color="#6666ff" stop-opacity="0.8"/>
  </linearGradient>
</defs>
3

添加交互功能

使用JavaScript和CSS实现交互功能。为每个原色圆添加拖拽控制点，当用户拖动控制点时，实时计算颜色混合结果并更新中心圆的颜色。同时添加连接线，显示每个原色圆对混合结果的影响。

<!-- JavaScript交互功能 -->
<script>
  // 拖拽功能
  function setupDrag(element, color) {
    let isDragging = false;

    element.addEventListener('mousedown', startDrag);
    document.addEventListener('mousemove', drag);
    document.addEventListener('mouseup', stopDrag);

    function startDrag(e) {
      isDragging = true;
      e.preventDefault();
    }

    function drag(e) {
      if (!isDragging) return;
      // 更新位置和颜色混合
      updateColorMix();
    }

    function stopDrag() {
      if (!isDragging) return;
      isDragging = false;
    }
  }

  // 更新颜色混合
  function updateColorMix() {
    const redIntensity = calculateIntensity(positions.red.x, positions.red.y);
    const greenIntensity = calculateIntensity(positions.green.x, positions.green.y);
    const blueIntensity = calculateIntensity(positions.blue.x, positions.blue.y);

    const mixedColor = `rgb(${Math.round(redIntensity)}, ${Math.round(greenIntensity)}, ${Math.round(blueIntensity)})`;
    mixCircle.setAttribute('fill', mixedColor);
  }
</script>
4

添加罗马数字标签和动画效果

为三个原色圆添加罗马数字标签（I、II、III），并使用CSS添加动画效果。包括颜色脉动动画、混合圆脉冲效果以及悬停交互效果，使整个图形更加生动和直观。

I
II
III
<!-- 添加罗马数字标签 -->
<text x="150" y="125" text-anchor="middle" font-family="'Times New Roman', serif" font-size="18" fill="#fff" font-weight="bold">I</text>
<text x="150" y="285" text-anchor="middle" font-family="'Times New Roman', serif" font-size="18" fill="#fff" font-weight="bold">II</text>
<text x="450" y="205" text-anchor="middle" font-family="'Times New Roman', serif" font-size="18" fill="#fff" font-weight="bold">III</text>

<!-- CSS动画效果 -->
<style>
  /* 混合圆脉冲动画 */
  #mixCircle {
    transition: all 0.5s ease;
    animation: pulse 3s ease-in-out infinite;
  }

  @keyframes pulse {
    0%, 100% {
      filter: url(#glow);
      r: 80;
    }
    50% {
      filter: url(#glow) brightness(1.1);
      r: 82;
    }
  }

  /* 颜色圆脉动动画 */
  .color-control circle:first-child {
    animation: colorPulse 2s ease-in-out infinite;
  }

  @keyframes colorPulse {
    0%, 100% {
      filter: url(#glow);
    }
    50% {
      filter: url(#glow) brightness(1.2);
    }
  }
</style>
